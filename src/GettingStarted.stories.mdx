# Usage

```jsx
import React from 'react';
import ReactDOM from 'react-dom';
import { App as AppContainer, Button } from 'react-library';

const App = () => {
  return (
    <AppContainer>
      <Button isPrimary={true}>Hello World</Button>
    </AppContainer>
  );
};

ReactDOM.render(<App />, document.querySelector('#app'));
```
